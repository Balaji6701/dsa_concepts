import java.util.*;
public class Hello {

    public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		String[] sentence = sc.nextLine().strip().split(" ");
		int len = 0;
        for(String word:sentence){
            if(word.length()>=len){
                len = word.length();
            }else{
                System.out.println("NO");
                return;
            }
        }
        System.out.println("YES");
	}
}
